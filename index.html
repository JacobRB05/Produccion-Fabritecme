Ejemplo PyScript funcionando en GitHub Pages

from pyscript import Element # Función que calcula el doble def calcular_doble(x): return int(x) * 2 # Función que se ejecuta al hacer clic def ejecutar(event): valor = Element("numero").value resultado = calcular_doble(valor) Element("resultado").write(f"El doble es: {resultado}") # Conectar el botón con la función Element("btn_calcular").element.onclick = ejecutar
